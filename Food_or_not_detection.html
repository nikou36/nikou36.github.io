  DATA_DIR = "/content/training"

  ds = tf.keras.preprocessing.image_dataset_from_directory(
    DATA_DIR,
    shuffle = True,
    label_mode = mode,
    seed=123,
    image_size=(res, res),
    batch_size=batch_size)
  return ds
